		}
	}

country_event = {
	id = ME_Byzantium_Events.2
	title = ME_Byzantium_Events.EVTNAME2
	desc = ME_Byzantium_Events.EVTDESC2
	picture = RELIGION_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = ME_Byzantium_Events2.EVTOPT1
		
		ai_chance = {
			factor = 0.15
			
			modifier = {
				factor = 0
				OR = {
					is_papal_controller = yes
					tag = PAP
					tag = SPA
					tag = CAS
					tag = ARA
					tag = HAB
					is_emperor = yes
					is_in_war = {
						attackers = ROOT
						casus_belli = cb_crusade
					}
				}
			}
			modifier = {
				factor = 5
				is_excommunicated = yes
			}
		}
		
		change_religion = orthodox
		if = {
			limit = {
				NOT = { religion = orthodox }
			}
			custom_tooltip = BYZ_THIS_WILL_APPLY_TO_PROVINCES
		}
		add_stability = -2
		add_prestige = -100
		add_legitimacy_equivalent = { value = -50 }
		hidden_effect = {
			every_owned_province = {
				limit = {
					is_core = ROOT
					religion = catholic
				}
				change_religion = orthodox
			}
		}
		capital_scope = {
			catholic_rebels = 2
		}
	}
	
	option = {
		name = ME_Byzantium_Events2.EVTOPT2
		
		ai_chance = {
			factor = 0.85
			
			modifier = {
				factor = 100
				OR = {
					is_papal_controller = yes
					tag = PAP
					tag = SPA
					tag = CAS
					tag = ARA
					tag = HAB
					is_emperor = yes
					is_in_war = {
						attackers = ROOT
						casus_belli = cb_crusade
					}
				}
			}
		}
		
		reverse_add_casus_belli = {
			target = FROM
			type = cb_annex
		}
		reverse_add_opinion = {
			who = FROM
			modifier = byz_religious_schism
		}
	}
}

province_event = {
	id = ME_Byzantium_Events.3
	title = ME_Byzantium_Events.EVTNAME3
	desc = ME_Byzantium_Events.EVTDESC3
	picture = CITY_VIEW_eventPicture
	
	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		province_id = 320
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = { province_has_center_of_trade_of_level = 1 }
				}
				center_of_trade = 2
			}
			else_if = {
				limit = {
					NOT = { province_has_center_of_trade_of_level = 3 }
				}
				add_center_of_trade_level = 2
			}
			add_trade_modifier = {
				who = FROM
				power = 25
				duration = -1
				key = byz_colossus_of_rhodes_trade_modifier
			}
		}
	}
	
	option = {
		name = ME_Byzantium_Events3.EVTOPT1
		
		ai_chance = { factor = 100 }
	}
}
