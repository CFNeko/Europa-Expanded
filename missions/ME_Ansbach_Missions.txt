			OR = {
				is_emperor = yes
				emperor = {

						who = ROOT

				}
			}
		}
		effect = {
			if = {
				limit = {
					is_elector = no

					}
				}

					elector = no
				}
			}
			else = {
				add_prestige = 10
			}
			add_adm_power = 50
			add_legitimacy_equivalent = { value = 5 }
		}
	}
