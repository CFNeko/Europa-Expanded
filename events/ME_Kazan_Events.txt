country_event = { #question of perm
	id = ME_Kazan_Events.4
	title = ME_Kazan_Events.4.t
	desc = ME_Kazan_Events.4.d
	picture = BORDER_TENSION_eventPicture

	fire_only_once = yes
	is_triggered_only = yes
	
	option = {
		name = ME_Kazan_Events.4.a
		
		trigger = {
			exists = PRM
		}
		
		ai_chance = { factor = 0.5 }
		
		PRM = {
			every_owned_province = {
				cede_province = ROOT
				add_province_modifier = {
					name = kaz_perm
					duration = 9125
				}
			}
		}
	}
	
	option = {
		name = ME_Kazan_Events.4.b
		
		trigger = {
			exists = PRM
			overlord_of = PRM
		}
		
		ai_chance = { factor = 0.1 }
		
		PRM = {
			grant_independence = yes
		}
		create_alliance = PRM
	}
	
	option = {
		name = ME_Kazan_Events.4.c
		
		trigger = {
			exists = PRM
			PRM = {
				OR = {
					is_subject = no
					is_subject_of = ROOT
				}
			}
		}
		
		ai_chance = { factor = 0.25 }
		
		create_subject = {
			subject = PRM
			subject_type = march
		}
	}
	
	option = {
		name = ME_Kazan_Events.4.d
		
		trigger = {
			exists = PRM
			PRM = {
				OR = {
					is_subject = no
					is_subject_of = ROOT
				}
			}
		}
		
		ai_chance = { factor = 0.15 }
		
		create_subject = {
			subject = PRM
			subject_type = tributary_state
		}
	}
	
	#in case perm doesnt exist
	option = {
		name = ME_Kazan_Events.4.d
		
		trigger = {
			NOT = { exists = PRM }
		}
		
		ai_chance = { factor = 1 }
		
		perm_area = {
			add_devastation = 10
		}
		add_years_of_income = 0.5
	}
}
